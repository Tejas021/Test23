public static JsonObject findFolderAndAppendDocument(JsonObject toc, String path, JsonObject newDocument) { // Split the folder path into individual folder names String[] folderNames = path.split("/"); System.out.println(path.isEmpty()); // Start with the root TOC object JsonObject currentFolder = toc; // Base case: If the folderNames array is empty, we've reached the target folder if (path.isEmpty()) { // Ensure the current folder has a 'documents' array System.out.println("Caleld"); if (!currentFolder.has("documents")) { currentFolder.add("documents", new JsonArray()); } // Append the new document to the 'documents' array currentFolder.getAsJsonArray("documents").add(newDocument); return currentFolder; } // Get the first folder name in the path String firstFolderName = folderNames[0]; System.out.println(firstFolderName); // Check if the current folder has subfolders if (currentFolder.has("subfolders") && currentFolder.get("subfolders").isJsonArray()) { JsonArray subfolders = currentFolder.getAsJsonArray("subfolders"); // Iterate through subfolders to find a match for (JsonElement subfolderElement : subfolders) { JsonObject subfolder = subfolderElement.getAsJsonObject(); System.out.println(subfolder); // Check if the current subfolder matches the first folder name if (subfolder.has("toc_name") && subfolder.get("toc_name").getAsString().equals(firstFolderName)) { // Recursively search for the rest of the path in the subfolder System.out.println("found"); String remainingPath = String.join("/", Arrays.copyOfRange(folderNames, 1, folderNames.length)); JsonObject updatedSubfolder = findFolderAndAppendDocument(subfolder, remainingPath, newDocument); if (updatedSubfolder != null) { subfolders.set(subfolders.indexOf(subfolderElement), updatedSubfolder); } // Return the entire updated TOC structure return toc; } } } // If no match is found at this level, the folder path is invalid return null; }
